function fitListNone() {
	function _show_next(hope, real, list, index) {
                    if(list.length < index) {
                        return;
                    }

		if(real() >= hope + 30) {
			list.eq(index - 1).hide();
			return;
		}

		list.eq(index).show();
		_show_next(hope, real, list, ++index);
	}

	var $target = $('#item_data_rec'),
		$items = $target.find('li'),
		tarWidth = $target.width(),
		tarHeight = $target.height(),
		itemWidth = tarWidth / Math.ceil(tarWidth / 332) - 10,
		itemHeight = 0,
		// 开发在1920宽度的屏幕上进行
		ratio = $(window).width() / 1920;

	$('.nonemsg').css('font-size', 45 * ratio + 'px');

	$items.each(function(i, o) {
		var $item = $(this),
			imgW = itemWidth - 2,
			imgH = imgW / 16 * 9,
			icoTop = imgH / 2 - 35,
			icoLeft = imgW / 2 - 35;

		$item.width(itemWidth);
		$item.find('img').width(imgW).height(imgH);
		$item.find('.shadow').width(imgW).height(imgH);
		$item.find('.iconPlay').css({ top: icoTop, left: icoLeft });
	});

	$items.hide();

	if(tarHeight > itemHeight) {
		_show_next($items.outerHeight(), function() {
			return $target.height();
		}, $items, 0);
	}
}

                target: '.tag-more-btn',

            this.$target = this.$wrapBox.find(this.config.target);
            this.itemHeight = this.$box.find('li').outerHeight(true);
            this.boxHeight();

        },
        boxHeight: function(){
            if(this.$box.find('ul').height()>this.itemHeight * 2){
                this.$target.show();
                this.$box.css({
                 'height': this.$target.hasClass(this.config.targetClass) ? this.$box.find('ul').outerHeight(true) : (this.itemHeight * 2)
                })
            }else{
                this.$target.hide();
                this.$box.css({
                 'height': 'auto'
                })
            }
        },
        renderHeight: function() {
            this.$box.animate({
                'height': this.itemHeight * 2
            });
            this.$target.removeClass(this.config.targetClass).html('展开');
        },

        autoHegith: function(height) {
            this.$box.animate({
                'height': height
            });
            this.$target.addClass(this.config.targetClass).html('收起');
        },

        bindEvent: function() {
            var _this = this;

            this.$target.click(function() {
                if ( $(this).hasClass(_this.config.targetClass) ) {
                    _this.renderHeight();
                } else {
                    _this.autoHegith( _this.$box.find('ul').outerHeight(true) );
                }
            });

            this.$wrapBox.on('click', 'li', function() {
                $(this).addClass('on').siblings('li').removeClass('on');
            });

            $(window).resize(function(event) {
                _this.boxHeight();
            });
        }
    };

    toggleTag.init();
})
