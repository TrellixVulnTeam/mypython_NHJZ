function updateMsg(id,data){
	var obj = $("#"+id);
	for(i=0;i<data.length;i++){
		if(i>4){
			break;
		}
		var json = data[i];
		var str = json.title;
		var id = json.id;
		if(str.length>11){
			str = "<a href=\"javascript:JsMod('/njlgdx/ggly/ggly_show?ggid="+id+"',600,400)\">"+str.substr(0,11)+"...</a>";
		}
		str += "<br />";
		obj.append(str);
	}
}
